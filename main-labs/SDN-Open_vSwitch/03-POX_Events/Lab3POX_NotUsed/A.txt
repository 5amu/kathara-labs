from pox.core import core

class A (EventMixin):

	_eventMixin_events = set([ pktInSeen ])

	def __init__ (self):
      		core.openflow.addListeners(self)
      		
      	def _handle_packetIn (self, event):

      		self.raiseEvent(pktInSeen("Generic"))

class pktInSeen (Event):

	def __init__ (self):
		Event.__init__(self)
     
def launch ( ):
	comp = A()
	core.register("A", comp)
